# 4.1 类的基本概念

> 原文：[C++ Classes](https://en.cppreference.com/w/cpp/language/classes)



类是C++面向对象编程的基础。类是一种用户定义的数据类型，它封装了数据（成员变量）和操作这些数据的函数（成员函数）。类是对象的蓝图，对象是类的实例。

## 类的基本结构

### 类的定义

```cpp
#include <iostream>
#include <string>

// 定义一个简单的类
class Person {
private:
    // 私有成员变量
    std::string name;
    int age;
    
public:
    // 公有成员函数
    void setName(const std::string& n) {
        name = n;
    }
    
    void setAge(int a) {
        age = a;
    }
    
    std::string getName() const {
        return name;
    }
    
    int getAge() const {
        return age;
    }
    
    void display() const {
        std::cout << "Name: " << name << ", Age: " << age << std::endl;
    }
};

int main() {
    // 创建对象
    Person person1;
    
    // 使用成员函数
    person1.setName("Alice");
    person1.setAge(25);
    person1.display();
    
    return 0;
}
```

### 访问修饰符

```cpp
class Example {
public:
    // 公有成员：可以在类外部访问
    int publicVar;
    void publicFunction() {
        std::cout << "Public function" << std::endl;
    }
    
private:
    // 私有成员：只能在类内部访问
    int privateVar;
    void privateFunction() {
        std::cout << "Private function" << std::endl;
    }
    
protected:
    // 受保护成员：只能在类内部和派生类中访问
    int protectedVar;
    void protectedFunction() {
        std::cout << "Protected function" << std::endl;
    }
};

int main() {
    Example obj;
    
    obj.publicVar = 10;        // 正确
    obj.publicFunction();      // 正确
    
    // obj.privateVar = 20;    // 错误：私有成员
    // obj.privateFunction();  // 错误：私有成员
    // obj.protectedVar = 30;  // 错误：受保护成员
    
    return 0;
}
```

## 构造函数和析构函数

### 构造函数

```cpp
class Rectangle {
private:
    double width;
    double height;
    
public:
    // 默认构造函数
    Rectangle() {
        width = 0.0;
        height = 0.0;
        std::cout << "Default constructor called" << std::endl;
    }
    
    // 参数化构造函数
    Rectangle(double w, double h) {
        width = w;
        height = h;
        std::cout << "Parameterized constructor called" << std::endl;
    }
    
    // 复制构造函数
    Rectangle(const Rectangle& other) {
        width = other.width;
        height = other.height;
        std::cout << "Copy constructor called" << std::endl;
    }
    
    // 成员函数
    double getArea() const {
        return width * height;
    }
    
    void display() const {
        std::cout << "Width: " << width << ", Height: " << height 
                  << ", Area: " << getArea() << std::endl;
    }
};

int main() {
    Rectangle rect1;           // 调用默认构造函数
    Rectangle rect2(5.0, 3.0); // 调用参数化构造函数
    Rectangle rect3(rect2);    // 调用复制构造函数
    
    rect1.display();
    rect2.display();
    rect3.display();
    
    return 0;
}
```

### 析构函数

```cpp
class Resource {
private:
    int* data;
    
public:
    // 构造函数
    Resource(int size) {
        data = new int[size];
        std::cout << "Resource allocated" << std::endl;
    }
    
    // 析构函数
    ~Resource() {
        delete[] data;
        std::cout << "Resource deallocated" << std::endl;
    }
    
    void setData(int index, int value) {
        data[index] = value;
    }
    
    int getData(int index) const {
        return data[index];
    }
};

int main() {
    {
        Resource res(5);
        res.setData(0, 10);
        res.setData(1, 20);
        
        std::cout << "Data[0] = " << res.getData(0) << std::endl;
        std::cout << "Data[1] = " << res.getData(1) << std::endl;
    } // 析构函数在这里被调用
    
    std::cout << "Resource object destroyed" << std::endl;
    
    return 0;
}
```

## 成员函数的定义

### 内联函数

```cpp
class Calculator {
private:
    double result;
    
public:
    // 内联成员函数
    inline void setResult(double r) {
        result = r;
    }
    
    inline double getResult() const {
        return result;
    }
    
    // 在类内部定义的函数自动内联
    void add(double value) {
        result += value;
    }
    
    void subtract(double value) {
        result -= value;
    }
    
    void multiply(double value) {
        result *= value;
    }
    
    void divide(double value) {
        if (value != 0) {
            result /= value;
        }
    }
    
    void display() const {
        std::cout << "Result: " << result << std::endl;
    }
};

int main() {
    Calculator calc;
    
    calc.setResult(10);
    calc.add(5);
    calc.multiply(2);
    calc.subtract(3);
    calc.divide(4);
    
    calc.display();
    
    return 0;
}
```

## 对象数组

```cpp
class Student {
private:
    std::string name;
    int score;
    
public:
    Student() : name(""), score(0) {}
    
    Student(const std::string& n, int s) : name(n), score(s) {}
    
    void setName(const std::string& n) {
        name = n;
    }
    
    void setScore(int s) {
        score = s;
    }
    
    std::string getName() const {
        return name;
    }
    
    int getScore() const {
        return score;
    }
    
    void display() const {
        std::cout << "Name: " << name << ", Score: " << score << std::endl;
    }
};

int main() {
    // 创建对象数组
    Student students[3];
    
    // 初始化数组
    students[0] = Student("Alice", 85);
    students[1] = Student("Bob", 92);
    students[2] = Student("Charlie", 78);
    
    // 显示所有学生信息
    std::cout << "Student Information:" << std::endl;
    for (int i = 0; i < 3; i++) {
        students[i].display();
    }
    
    // 计算平均分
    double total = 0;
    for (int i = 0; i < 3; i++) {
        total += students[i].getScore();
    }
    double average = total / 3;
    
    std::cout << "Average score: " << average << std::endl;
    
    return 0;
}
```

## 类的设计原则

### 1. 封装

```cpp
class BankAccount {
private:
    double balance;
    std::string accountNumber;
    
public:
    BankAccount(const std::string& accNum, double initialBalance) 
        : accountNumber(accNum), balance(initialBalance) {}
    
    // 公共接口
    bool deposit(double amount) {
        if (amount > 0) {
            balance += amount;
            return true;
        }
        return false;
    }
    
    bool withdraw(double amount) {
        if (amount > 0 && amount <= balance) {
            balance -= amount;
            return true;
        }
        return false;
    }
    
    double getBalance() const {
        return balance;
    }
    
    std::string getAccountNumber() const {
        return accountNumber;
    }
};
```

### 2. 数据隐藏

```cpp
class Temperature {
private:
    double celsius;
    
public:
    Temperature(double c) : celsius(c) {}
    
    // 通过公共接口访问私有数据
    double getCelsius() const {
        return celsius;
    }
    
    double getFahrenheit() const {
        return celsius * 9.0 / 5.0 + 32.0;
    }
    
    void setCelsius(double c) {
        celsius = c;
    }
    
    void setFahrenheit(double f) {
        celsius = (f - 32.0) * 5.0 / 9.0;
    }
};
```

## 如何使它崩溃

尝试以下操作来观察程序的行为：

1. **未初始化的对象**：
```cpp
class Test {
private:
    int* ptr;
public:
    void usePtr() {
        *ptr = 42;  // 使用未初始化的指针
    }
};

int main() {
    Test obj;
    obj.usePtr();  // 程序崩溃
    return 0;
}
```

2. **内存泄漏**：
```cpp
class BadClass {
private:
    int* data;
public:
    BadClass() {
        data = new int[100];
    }
    // 缺少析构函数，导致内存泄漏
};

int main() {
    BadClass obj;
    return 0;  // 内存泄漏
}
```

3. **访问私有成员**：
```cpp
class PrivateClass {
private:
    int secret;
public:
    PrivateClass() : secret(42) {}
};

int main() {
    PrivateClass obj;
    // std::cout << obj.secret << std::endl;  // 编译错误
    return 0;
}
```
