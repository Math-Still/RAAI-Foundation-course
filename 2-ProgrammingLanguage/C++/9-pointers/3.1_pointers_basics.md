# 3.1 指针基础

> 原文：[C++ Pointers](https://en.cppreference.com/w/cpp/language/pointer)



指针是C++中最强大但也最复杂的特性之一。指针是一个变量，它存储另一个变量的内存地址。通过指针，我们可以直接访问和操作内存，这使得C++能够进行底层编程。

## 指针的基本概念

### 指针的声明和初始化

```cpp
#include <iostream>

int main() {
    int number = 42;
    
    // 声明指针
    int* ptr;  // ptr是一个指向int的指针
    
    // 初始化指针
    ptr = &number;  // &运算符获取变量的地址
    
    std::cout << "number的值: " << number << std::endl;
    std::cout << "number的地址: " << &number << std::endl;
    std::cout << "ptr存储的地址: " << ptr << std::endl;
    std::cout << "ptr指向的值: " << *ptr << std::endl;  // *运算符解引用
    
    return 0;
}
```

### 指针的基本操作

```cpp
int main() {
    int x = 10;
    int y = 20;
    
    // 声明和初始化指针
    int* ptr1 = &x;
    int* ptr2 = &y;
    
    std::cout << "x = " << x << ", y = " << y << std::endl;
    std::cout << "ptr1指向: " << *ptr1 << std::endl;
    std::cout << "ptr2指向: " << *ptr2 << std::endl;
    
    // 通过指针修改值
    *ptr1 = 100;
    *ptr2 = 200;
    
    std::cout << "修改后:" << std::endl;
    std::cout << "x = " << x << ", y = " << y << std::endl;
    std::cout << "ptr1指向: " << *ptr1 << std::endl;
    std::cout << "ptr2指向: " << *ptr2 << std::endl;
    
    return 0;
}
```

## 空指针

```cpp
#include <iostream>

int main() {
    // 空指针的几种表示方法
    int* ptr1 = nullptr;  // C++11推荐
    int* ptr2 = NULL;     // C风格
    int* ptr3 = 0;        // 字面量0
    
    std::cout << "ptr1: " << ptr1 << std::endl;
    std::cout << "ptr2: " << ptr2 << std::endl;
    std::cout << "ptr3: " << ptr3 << std::endl;
    
    // 检查空指针
    if (ptr1 == nullptr) {
        std::cout << "ptr1是空指针" << std::endl;
    }
    
    // 危险：解引用空指针
    // std::cout << *ptr1 << std::endl;  // 未定义行为
    
    return 0;
}
```

## 指针和数组

```cpp
#include <iostream>

int main() {
    int arr[] = {1, 2, 3, 4, 5};
    int* ptr = arr;  // 数组名是指向第一个元素的指针
    
    std::cout << "通过数组访问:" << std::endl;
    for (int i = 0; i < 5; i++) {
        std::cout << arr[i] << " ";
    }
    std::cout << std::endl;
    
    std::cout << "通过指针访问:" << std::endl;
    for (int i = 0; i < 5; i++) {
        std::cout << *(ptr + i) << " ";
    }
    std::cout << std::endl;
    
    std::cout << "通过指针下标访问:" << std::endl;
    for (int i = 0; i < 5; i++) {
        std::cout << ptr[i] << " ";
    }
    std::cout << std::endl;
    
    return 0;
}
```

## 指针的算术运算

```cpp
#include <iostream>

int main() {
    int arr[] = {10, 20, 30, 40, 50};
    int* ptr = arr;
    
    std::cout << "原始指针: " << ptr << std::endl;
    std::cout << "指向的值: " << *ptr << std::endl;
    
    // 指针递增
    ptr++;
    std::cout << "递增后指针: " << ptr << std::endl;
    std::cout << "指向的值: " << *ptr << std::endl;
    
    // 指针递减
    ptr--;
    std::cout << "递减后指针: " << ptr << std::endl;
    std::cout << "指向的值: " << *ptr << std::endl;
    
    // 指针加法
    ptr = ptr + 2;
    std::cout << "加2后指针: " << ptr << std::endl;
    std::cout << "指向的值: " << *ptr << std::endl;
    
    // 指针减法
    ptr = ptr - 1;
    std::cout << "减1后指针: " << ptr << std::endl;
    std::cout << "指向的值: " << *ptr << std::endl;
    
    // 指针相减
    int* ptr2 = arr + 4;
    std::cout << "ptr2指向: " << *ptr2 << std::endl;
    std::cout << "ptr2 - ptr = " << ptr2 - ptr << std::endl;
    
    return 0;
}
```

## 指针和函数

```cpp
#include <iostream>

// 通过指针传递参数
void modifyValue(int* ptr) {
    *ptr = 100;
}

// 通过指针返回多个值
void getMinMax(int arr[], int size, int* min, int* max) {
    *min = arr[0];
    *max = arr[0];
    
    for (int i = 1; i < size; i++) {
        if (arr[i] < *min) {
            *min = arr[i];
        }
        if (arr[i] > *max) {
            *max = arr[i];
        }
    }
}

// 返回指针
int* findElement(int arr[], int size, int value) {
    for (int i = 0; i < size; i++) {
        if (arr[i] == value) {
            return &arr[i];
        }
    }
    return nullptr;
}

int main() {
    int number = 42;
    std::cout << "原始值: " << number << std::endl;
    
    modifyValue(&number);
    std::cout << "修改后: " << number << std::endl;
    
    int arr[] = {5, 2, 8, 1, 9, 3};
    int size = 6;
    int min, max;
    
    getMinMax(arr, size, &min, &max);
    std::cout << "最小值: " << min << ", 最大值: " << max << std::endl;
    
    int* found = findElement(arr, size, 8);
    if (found != nullptr) {
        std::cout << "找到元素8，位置: " << (found - arr) << std::endl;
    } else {
        std::cout << "未找到元素8" << std::endl;
    }
    
    return 0;
}
```

## 指针的常见错误

### 1. 未初始化的指针

```cpp
int main() {
    int* ptr;  // 未初始化
    // std::cout << *ptr << std::endl;  // 未定义行为
    return 0;
}
```

### 2. 悬空指针

```cpp
int main() {
    int* ptr = new int(42);
    delete ptr;
    // std::cout << *ptr << std::endl;  // 悬空指针
    ptr = nullptr;  // 正确的做法
    return 0;
}
```

### 3. 类型不匹配

```cpp
int main() {
    int number = 42;
    double* ptr = &number;  // 类型不匹配
    return 0;
}
```

## 如何使它崩溃

尝试以下操作来观察程序的行为：

1. **解引用空指针**：
```cpp
int main() {
    int* ptr = nullptr;
    *ptr = 42;  // 程序崩溃
    return 0;
}
```

2. **越界访问**：
```cpp
int main() {
    int arr[5] = {1, 2, 3, 4, 5};
    int* ptr = arr;
    std::cout << *(ptr + 10) << std::endl;  // 越界访问
    return 0;
}
```

3. **使用已释放的内存**：
```cpp
int main() {
    int* ptr = new int(42);
    delete ptr;
    *ptr = 100;  // 使用已释放的内存
    return 0;
}
```
